/** * P2PPong v5.0 — Бронебойная сигнальная машина * Чистый модуль. Ноль зависимостей от DOM, localStorage, интерфейса. * Работает везде: browser, Web Worker, отдельная страница, встроенный модуль. * * Раньше назывался Пинбол. */ const P2PPong = (() => { 'use strict'; // ==================== КОНФИГУРАЦИЯ ==================== const SERVERS = [ { id: 'render', url: 'https://p2ppong-v2.onrender.com' }, { id: 'workers', url: 'https://robincall.stephanclaps-491.workers.dev' } ]; // ==================== СОСТОЯНИЕ ==================== let channels = {}; // { sessionId: { status, sessionKey, server, peerId, allServers } } let pollIntervals = {}; // { sessionId: intervalId } let callbacks = {}; // { onReady, onChannelOpen, onMessage, onError, onSessionCreated } let peerId = null; // Загружается асинхронно из IndexedDB let sessionId = null; // Уникальный ID этой сессии (в памяти) let isInitialized = false; // ==================== УТИЛИТЫ ==================== function generateToken() { return crypto.randomUUID?.() || Math.random().toString(36).substring(2) + Date.now().toString(36); } async function sha256(text) { const data = new TextEncoder().encode(text); const hash = await crypto.subtle.digest('SHA-256', data); return Array.from(new Uint8Array(hash)) .map(b => b.toString(16).padStart(2, '0')).join(''); } function log(msg) { console.log('[P2PPong]', msg); } // ==================== ХРАНИЛИЩЕ (IndexedDB, без localStorage) ==================== async function openDB() { return new Promise((resolve, reject) => { const req = indexedDB.open('P2PPongSecure', 1); req.onupgradeneeded = () => { if (!req.result.objectStoreNames.contains('identity')) req.result.createObjectStore('identity', { keyPath: 'id' }); }; req.onsuccess = () => resolve(req.result); req.onerror = () => reject(req.error); }); } async function getOrCreatePeerId() { if (peerId) return peerId; try { const db = await openDB(); const tx = db.transaction('identity', 'readonly'); const store = tx.objectStore('identity'); const stored = await new Promise((resolve, reject) => { const req = store.get('peerId'); req.onsuccess = () => resolve(req.result); req.onerror = () => reject(req.error); }); if (stored?.value) { peerId = stored.value; return peerId; } } catch (e) { log('IndexedDB недоступна, генерируем временный peerId'); } // Если IndexedDB не сработала — временный peerId на сессию peerId = generateToken().substring(0, 16); return peerId; } async function savePeerId() { try { const db = await openDB(); const tx = db.transaction('identity', 'readwrite'); tx.objectStore('identity').put({ id: 'peerId', value: peerId }); } catch (e) { log('Не удалось сохранить peerId'); } } // ==================== AES-GCM ШИФРОВАНИЕ ==================== async function deriveKey(secret) { const hash = await sha256(secret); const keyBytes = new TextEncoder().encode(hash.substring(0, 32)); return await crypto.subtle.importKey( 'raw', keyBytes, { name: 'AES-GCM' }, false, ['encrypt', 'decrypt'] ); } async function encryptPacket(plaintext, secret) { try { const key = await deriveKey(secret); const iv = crypto.getRandomValues(new Uint8Array(12)); const encoded = new TextEncoder().encode(plaintext); const ciphertext = await crypto.subtle.encrypt({ name: 'AES-GCM', iv }, key, encoded); const combined = new Uint8Array(iv.length + new Uint8Array(ciphertext).length); combined.set(iv); combined.set(new Uint8Array(ciphertext), iv.length); return '🔐' + btoa(String.fromCharCode(...combined)); } catch (e) { return plaintext; } } async function decryptPacket(encryptedText, secret) { if (!encryptedText || !encryptedText.startsWith('🔐')) return null; try { const key = await deriveKey(secret); const combined = Uint8Array.from(atob(encryptedText.substring(2)), c => c.charCodeAt(0)); const iv = combined.slice(0, 12); const ct = combined.slice(12); const decrypted = await crypto.subtle.decrypt({ name: 'AES-GCM', iv }, key, ct); return new TextDecoder().decode(decrypted); } catch (e) { return null; } } // ==================== СЕТЕВЫЕ ЗАПРОСЫ ==================== async function apiCall(serverId, method, path, body = null) { const server = SERVERS.find(s => s.id === serverId); if (!server) return null; const ctrl = new AbortController(); const timeout = setTimeout(() => ctrl.abort(), serverId === 'workers' ? 5000 : 10000); try { const opts = { method, headers: { 'Content-Type': 'application/json' }, signal: ctrl.signal }; if (body) opts.body = JSON.stringify(body); const res = await fetch(server.url + path, opts); clearTimeout(timeout); if (!res.ok) return null; return await res.json(); } catch (e) { clearTimeout(timeout); return null; } } // ==================== ПОЛЛИНГ ==================== function startPolling(sid, serverId) { if (pollIntervals[sid]) return; let lastTime = Date.now(); const poll = async () => { const srv = serverId || channels[sid]?.server; const res = await apiCall(srv, 'GET', '/message?id=' + sid + '&since=' + lastTime); if (!res?.messages) return; for (const m of res.messages) { const ch = channels[sid]; if (!ch) continue; // Пробуем расшифровать let decrypted = await decryptPacket(m.packet, ch.sessionSecret); // Если не расшифровалось — пробуем как голый JSON (обратная совместимость) if (!decrypted) { try { const parsed = JSON.parse(m.packet); decrypted = parsed.text || m.packet; } catch (e) { decrypted = m.packet; } } // Извлекаем отправителя let sender = null; let text = decrypted; try { const parsed = JSON.parse(decrypted); sender = parsed.sender; text = parsed.text || decrypted; } catch (e) {} // Фильтруем свои сообщения if (sender === sessionId) { const msgTime = m.time > 9999999999 ? m.time : m.time * 1000; lastTime = Math.max(lastTime, msgTime); continue; } // Вызываем колбэк if (callbacks.onMessage) { callbacks.onMessage({ text, channelId: sid, sender }); } const msgTime = m.time > 9999999999 ? m.time : m.time * 1000; lastTime = Math.max(lastTime, msgTime); } }; poll(); pollIntervals[sid] = setInterval(poll, 4000); } function stopPolling(sid) { if (pollIntervals[sid]) { clearInterval(pollIntervals[sid]); delete pollIntervals[sid]; } } function stopAllPolling() { Object.keys(pollIntervals).forEach(stopPolling); } // ==================== ОСНОВНЫЕ МЕТОДЫ ==================== /** * Создать стрелу (маяк) * Вызывает колбэки: onSessionCreated, onChannelOpen */ async function createArrow() { await ensureInit(); const tempKey = generateToken(); const tempKeyHash = await sha256(tempKey); const sessionSecret = tempKey; // Для AES-GCM используем tempKey как секрет log('Создание маяка...'); const results = []; for (const server of SERVERS) { const body = server.id === 'workers' ? { tempKeyHash, publicId: peerId } : { tempKey: tempKey, tempKeyHash, publicId: peerId }; const res = await apiCall(server.id, 'POST', '/beacon', body); if (res?.sessionId) { results.push({ sessionId: res.sessionId, server: server.id }); log('Маяк на ' + server.id + ': ' + res.sessionId.substring(0, 8)); } } if (results.length === 0) { if (callbacks.onError) callbacks.onError({ text: 'Все серверы не отвечают' }); return; } const primary = results[0]; const sid = primary.sessionId; channels[sid] = { status: 'waiting', sessionSecret, server: primary.server, peerId: peerId, allServers: results.map(r => r.server) }; const serversParam = results.map(r => r.server).join(','); const fullLink = `https://stepweather-prog.github.io/ROBINHOOD-P2P/?id=${sid}&servers=${serversParam}&peer=${peerId}&key=${tempKey}`; if (callbacks.onSessionCreated) { callbacks.onSessionCreated({ fullLink, peerId, key: tempKey }); } // Ожидание matched let matched = false; const start = Date.now(); while (Date.now() - start < 90000) { const checks = await Promise.all( results.map(r => apiCall(r.server, 'GET', '/beacon?id=' + r.sessionId)) ); if (checks.some(c => c?.matched)) { matched = true; break; } await new Promise(r => setTimeout(r, 5000)); } if (!matched) { delete channels[sid]; if (callbacks.onError) callbacks.onError({ text: 'Таймаут ожидания' }); return; } channels[sid].status = 'open'; startPolling(sid, channels[sid].server); if (callbacks.onChannelOpen) { callbacks.onChannelOpen({ channelId: sid, role: 'creator', peerId }); } log('Канал открыт!'); } /** * Принять стрелу */ async function acceptArrow(peerIdFromLink, tempKey) { await ensureInit(); const keyHash = await sha256(tempKey); const sessionSecret = tempKey; log('Поиск маяка для пира ' + peerIdFromLink + '...'); // Пробуем новый метод for (const server of SERVERS) { try { const res = await apiCall(server.id, 'POST', '/find-by-peer', { targetPublicId: peerIdFromLink, myPublicId: peerId, tempKeyHash: keyHash }); if (res?.status === 'matched') { await openChannel(res.sessionId, server.id, sessionSecret, peerIdFromLink); return; } } catch (e) {} } // Fallback: старый метод const urlParams = new URLSearchParams( typeof window !== 'undefined' ? window.location.search : '' ); const serversParam = urlParams.get('servers') || 'render,workers'; const serversList = serversParam.split(','); const sid = urlParams.get('id'); for (const srvId of serversList) { const body = srvId === 'workers' ? { tempKeyHash: keyHash } : { tempKey }; const res = await apiCall(srvId, 'POST', '/find', body); if (res?.status === 'matched') { await openChannel(res.sessionId || sid, srvId, sessionSecret, peerIdFromLink); return; } } if (callbacks.onError) callbacks.onError({ text: 'Маяк не найден' }); } async function openChannel(sid, serverId, sessionSecret, peerIdFromLink) { channels[sid] = { status: 'open', sessionSecret, server: serverId, peerId: peerIdFromLink }; startPolling(sid, serverId); if (callbacks.onChannelOpen) { callbacks.onChannelOpen({ channelId: sid, role: 'receiver', peerId: peerIdFromLink }); } log('Подключено через ' + serverId + '!'); } /** * Отправить сообщение в канал */ async function sendMessage(channelId, text) { await ensureInit(); const ch = channels[channelId]; if (!ch) { if (callbacks.onError) callbacks.onError({ text: 'Канал не найден', channelId }); return; } const payload = JSON.stringify({ text, sender: sessionId, time: Date.now() }); const encrypted = await encryptPacket(payload, ch.sessionSecret); const res = await apiCall(ch.server, 'POST', '/message', { sessionId: channelId, packet: encrypted }); if (!res) { log('Ошибка отправки', 'error'); delete channels[channelId]; stopPolling(channelId); if (callbacks.onError) callbacks.onError({ text: 'session_not_found', channelId }); } } /** * Подписка на события */ function on(event, cb) { callbacks[event] = cb; } /** * Получить свой PeerId */ function getPeerId() { return peerId; } /** * Проверить статус серверов */ async function checkServers() { const statuses = {}; for (const server of SERVERS) { const ping = await apiCall(server.id, 'GET', '/ping'); statuses[server.id] = ping ? 'online' : 'offline'; } return statuses; } // ==================== ИНИЦИАЛИЗАЦИЯ ==================== async function ensureInit() { if (!isInitialized) await init(); } async function init() { if (isInitialized) return; sessionId = generateToken(); await getOrCreatePeerId(); await savePeerId(); log('P2PPong v5.0 запущен. PeerId: ' + peerId.substring(0, 10) + '...'); // Пингуем серверы const statuses = await checkServers(); const online = Object.values(statuses).some(s => s === 'online'); isInitialized = true; if (callbacks.onReady) { callbacks.onReady({ peerId, statuses, online }); } } // ==================== ЖИЗНЕННЫЙ ЦИКЛ ==================== function destroy() { stopAllPolling(); channels = {}; callbacks = {}; isInitialized = false; } // ==================== ПУБЛИЧНОЕ API ==================== return { // Основные методы init, createArrow, acceptArrow, sendMessage, // Подписки on, // Информация getPeerId, checkServers, // Жизненный цикл destroy, // Служебное (для обратной совместимости) getSessionId: () => sessionId }; })(); // ==================== АВТОЗАПУСК ==================== // Если модуль загружен в браузере — запускаемся if (typeof window !== 'undefined') { P2PPong.init().catch(e => console.error('[P2PPong] Init error:', e)); } // ==================== ЭКСПОРТ ==================== // Для использования как ES модуль if (typeof module !== 'undefined' && module.exports) { module.exports = P2PPong; }
